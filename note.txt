//---------------------- Abstraction with YouTube Videos ----------------------------

Video:
    // Member Variables
    _title: string
    _author: string
    _length: int (seconds)
    _comments: List<Comment>

    // Construtors
    Video(title, author, length)

    // Methods
    NumbersOfComments(): int

Comment:
    // Member Variables
    _name: string
    _text: string
